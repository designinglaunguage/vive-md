## 📁 파일 저장 구조

### 디렉토리 구조

```
docs/
└── news/
    └── kbw-weekly/               # KBW 주간브리프
        ├── 2026/
        │   ├── [주간브리프]KBW_260209-06.md
        │   ├── [주간브리프]KBW_260216-07.md
        │   └── ...
        ├── archive/              # 원문 보관
        │   └── raw-YYYY-MM-DD/
        └── templates/            # 템플릿 보관
            └── kbw-template.md
```

### 파일명 규칙

| 유형 | 파일명 패턴 | 예시 |
|------|------------|------|
| 주간브리프 | `[주간브리프]KBW_YYMMDD-NN.md` | `[주간브리프]KBW_260209-06.md` |
| 원문 보관 | `article-{lang}-##-raw.md` | `article-cn-01-raw.md` |

### 파일명 생성 규칙

- `YYMMDD`: 날짜 (예: 260209 = 2026년 2월 9일)
- `NN`: 버전/차수 (01부터 순차적)
- 이번 주에 이미 파일이 있으면 번호 증가

## 🤖 자동화 스크립트

### Python 스크립트 예시

```python
#!/usr/bin/env python3
"""
KBW 주간브리프 자동화 스크립트
"""

import os
import re
import glob
from datetime import datetime, timedelta
from pathlib import Path

# 설정
NEWS_DIR = Path("docs/news/kbw-weekly")
ARCHIVE_DIR = Path("docs/news/kbw-weekly/archive")

class KBWWeeklyReport:
    def __init__(self):
        self.today = datetime.now()
        self.year = self.today.strftime("%Y")
        self.year_short = self.today.strftime("%y")
        self.month = self.today.strftime("%m")
        self.day = self.today.strftime("%d")
        self.date_str = self.today.strftime("%y%m%d")  # YYMMDD
        
    def get_next_version(self):
        """다음 버전 번호 계산"""
        pattern = f"[주간브리프]KBW_{self.date_str}-*.md"
        existing_files = list(NEWS_DIR.glob(pattern))
        
        if not existing_files:
            return "01"
        
        versions = []
        for f in existing_files:
            match = re.search(rf'KBW_{self.date_str}-(\d+)\.md$', f.name)
            if match:
                versions.append(int(match.group(1)))
        
        next_version = max(versions) + 1 if versions else 1
        return f"{next_version:02d}"
    
    def generate_filename(self):
        """파일명 생성"""
        version = self.get_next_version()
        return f"[주간브리프]KBW_{self.date_str}-{version}.md"
    
    def generate_filepath(self):
        """파일 경로 생성"""
        year_dir = NEWS_DIR / self.year
        year_dir.mkdir(parents=True, exist_ok=True)
        return year_dir / self.generate_filename()
    
    def create_directory_structure(self):
        """디렉토리 구조 생성"""
        paths = [
            NEWS_DIR / self.year,
            ARCHIVE_DIR / f"raw-{self.today.strftime('%Y-%m-%d')}",
        ]
        for path in paths:
            path.mkdir(parents=True, exist_ok=True)
        return paths[0]
    
    def create_template(self):
        """KBW 주간브리프 템플릿 생성"""
        template = f"""※ 출처에 표기된 언론사를 클릭하시면 원문을 조회할 수 있습니다.

[정책/시장 동향]



[전방산업 동향]



[업체/기술 동향]



"""
        return template

if __name__ == "__main__":
    report = KBWWeeklyReport()
    report.create_directory_structure()
    filepath = report.generate_filepath()
    template = report.create_template()
    
    print(f"파일 경로: {filepath}")
    print(f"템플릿:\n{template}")
```

## 📊 품질 관리 자동화

### KBW 형식 검증 스크립트

```python
#!/usr/bin/env python3
"""
KBW 주간브리프 형식 검증 스크립트
"""

import re
from pathlib import Path

def validate_kbw_format(file_path: Path) -> dict:
    """KBW 주간브리프 형식 검증"""
    issues = []
    stats = {
        "total_articles": 0,
        "policy_market": 0,
        "oem": 0,
        "supplier_tech": 0,
        "sources": 0,
    }
    
    with open(file_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # 1. 헤더 확인
    if "※ 출처에 표기된 언론사를 클릭하시면 원문을 조회할 수 있습니다." not in content:
        issues.append("헤더 문구 누락")
    
    # 2. 섹션 확인
    required_sections = ["[정책/시장 동향]", "[전방산업 동향]", "[업체/기술 동향]"]
    for section in required_sections:
        if section not in content:
            issues.append(f"섹션 누락: {section}")
    
    # 3. 뉴스 항목 수 확인
    articles = re.findall(r'□ ', content)
    stats["total_articles"] = len(articles)
    
    if len(articles) < 12:
        issues.append(f"뉴스 건수 부족: {len(articles)}건 (최소 12건 권장)")
    if len(articles) > 18:
        issues.append(f"뉴스 건수 과다: {len(articles)}건 (최대 18건 권장)")
    
    # 4. 섹션별 건수 확인
    sections = content.split('[전방산업 동향]')
    if len(sections) >= 2:
        policy_section = sections[0]
        remaining = sections[1].split('[업체/기술 동향]')
        if len(remaining) >= 2:
            oem_section = remaining[0]
            tech_section = remaining[1]
            
            stats["policy_market"] = len(re.findall(r'□ ', policy_section))
            stats["oem"] = len(re.findall(r'□ ', oem_section))
            stats["supplier_tech"] = len(re.findall(r'□ ', tech_section))
    
    # 5. 출처 표기 확인
    sources = re.findall(r'☞ 출처:', content)
    stats["sources"] = len(sources)
    
    if len(sources) != len(articles):
        issues.append(f"출처 표기 불일치: 뉴스 {len(articles)}건, 출처 {len(sources)}건")
    
    # 6. 날짜 형식 확인
    date_pattern = r'\d{4}/\d{2}/\d{2}'
    dates = re.findall(date_pattern, content)
    if len(dates) != len(articles):
        issues.append(f"날짜 형식 불일치: 뉴스 {len(articles)}건, 날짜 {len(dates)}개")
    
    return {
        "valid": len(issues) == 0,
        "issues": issues,
        "stats": stats
    }

if __name__ == "__main__":
    import sys
    if len(sys.argv) > 1:
        result = validate_kbw_format(Path(sys.argv[1]))
        print(f"검증 결과: {'통과' if result['valid'] else '실패'}")
        if result['issues']:
            print("발견된 문제:")
            for issue in result['issues']:
                print(f"  - {issue}")
        print(f"\n통계: {result['stats']}")
```

## 📝 README 자동 업데이트

### 최신 KBW 주간브리프 목록 갱신

```python
#!/usr/bin/env python3
"""
README.md 최신 KBW 주간브리프 목록 자동 업데이트
"""

from pathlib import Path
from datetime import datetime

def update_kbw_readme():
    """KBW README 업데이트"""
    
    kbw_dir = Path("docs/news/kbw-weekly")
    readme_path = Path("docs/news/kbw-weekly/README.md")
    
    # 최신 10개 주간브리프 찾기
    weekly_reports = sorted(kbw_dir.rglob("[주간브리프]*.md"), reverse=True)[:10]
    
    # 새로운 목록 생성
    report_list = "## 📋 최신 KBW 주간브리프\n\n"
    for report in weekly_reports:
        # 파일명에서 날짜 추출
        match = re.search(r'KBW_(\d{6})-(\d+)', report.name)
        if match:
            date_str = match.group(1)
            version = match.group(2)
            formatted_date = f"20{date_str[:2]}년 {date_str[2:4]}월 {date_str[4:6]}일"
            report_list += f"- [{formatted_date} - {version}차](./{report.parent.name}/{report.name})\n"
    
    # README 업데이트
    if readme_path.exists():
        with open(readme_path, 'r', encoding='utf-8') as f:
            content = f.read()
        
        import re
        pattern = r'## 📋 최신 KBW 주간브리프.*?(?=## |\Z)'
        new_content = re.sub(pattern, report_list + "\n\n", content, flags=re.DOTALL)
        
        with open(readme_path, 'w', encoding='utf-8') as f:
            f.write(new_content)
        
        print("README.md 업데이트 완료")

if __name__ == "__main__":
    import re
    update_kbw_readme()
```

## 🐛 문제 해결

### 자주 발생하는 문제

| 문제 | 원인 | 해결 방법 |
|------|------|----------|
| 파일명 중복 | 동일 날짜에 여러 버전 | 자동으로 버전 번호 증가 |
| 섹션별 불균형 | 특정 섹션 뉴스 부족 | 다른 사이트 추가 검색 |
| 출처 누락 | 원문에 출처 불명확 | 원문 링크 직접 확인 |
| 형식 불일치 | 자동 변환 오류 | 수동 검증 후 수정 |
